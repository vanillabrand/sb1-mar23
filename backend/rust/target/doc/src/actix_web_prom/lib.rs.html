2Prometheus instrumentation for [actix-web](https://github.com/actix/actix-web). This middleware is heavily influenced by the work in [sd2k/rocket_prometheus](https://github.com/sd2k/rocket_prometheus). We track the same default metrics and allow for adding user defined metrics.
3
4By default two metrics are tracked (this assumes the namespace `actix_web_prom`):
5
6  - `actix_web_prom_http_requests_total` (labels: endpoint, method, status): the total number
7   of HTTP requests handled by the actix HttpServer.
8
9  - `actix_web_prom_http_requests_duration_seconds` (labels: endpoint, method, status): the
10   request duration for all HTTP requests handled by the actix HttpServer.
11

90## Custom metrics
91
92You instantiate `PrometheusMetrics` and then use its `.registry` to register your custom
93metric (in this case, we use a `IntCounterVec`).
94
95Then you can pass this counter through `.data()` to have it available within the resource
96responder.
97
98```rust
99use actix_web::{web, App, HttpResponse, HttpServer};
100use actix_web_prom::{PrometheusMetrics, PrometheusMetricsBuilder};
101use prometheus::{opts, IntCounterVec};
102
103async fn health(counter: web::Data<IntCounterVec>) -> HttpResponse {
104    counter.with_label_values(&["endpoint", "method", "status"]).inc();
105    HttpResponse::Ok().finish()
106}
107
108#[actix_web::main]
109async fn main() -> std::io::Result<()> {
110    let prometheus = PrometheusMetricsBuilder::new("api")
111        .endpoint("/metrics")
112        .build()
113        .unwrap();
114
115    let counter_opts = opts!("counter", "some random counter").namespace("api");
116    let counter = IntCounterVec::new(counter_opts, &["endpoint", "method", "status"]).unwrap();
117    prometheus
118        .registry
119        .register(Box::new(counter.clone()))
120        .unwrap();
121
122# if false {
123        HttpServer::new(move || {
124            App::new()
125                .wrap(prometheus.clone())
126                .data(counter.clone())
127                .service(web::resource("/health").to(health))
128        })
129        .bind("127.0.0.1:8080")?
130        .run()
131        .await?;
132# }
133    Ok(())
134}

236#[derive(Debug)]
237/// Builder to create new PrometheusMetrics struct.HistogramVec
238///
239/// It allows setting optional parameters like registry, buckets, etc.
240pub struct PrometheusMetricsBuilder {
241    namespace: String,
242    endpoint: Option<String>,
243    const_labels: HashMap<String, String>,
244    registry: Registry,
245    buckets: Vec<f64>,
246}
247
248impl PrometheusMetricsBuilder {
249    /// Create new PrometheusMetricsBuilder
250    ///
251    /// namespace example: "actix"
252    pub fn new(namespace: &str) -> Self {
253        Self {
254            namespace: namespace.into(),
255            endpoint: None,
256            const_labels: HashMap::new(),
257            registry: Registry::new(),
258            buckets: prometheus::DEFAULT_BUCKETS.to_vec(),
259        }
260    }
261
262    /// Set actix web endpoint
263    ///
264    /// Example: "/metrics"
265    pub fn endpoint(mut self, value: &str) -> Self {
266        self.endpoint = Some(value.into());
267        self
268    }
269
270    /// Set histogram buckets
271    pub fn buckets(mut self, value: &[f64]) -> Self {
272        self.buckets = value.to_vec();
273        self
274    }
275
276    /// Set labels to add on every metrics
277    pub fn const_labels(mut self, value: HashMap<String, String>) -> Self {
278        self.const_labels = value;
279        self
280    }
281
282    /// Set registry
283    ///
284    /// By default one is set and is internal to PrometheusMetrics
285    pub fn registry(mut self, value: Registry) -> Self {
286        self.registry = value;
287        self
288    }
289
290    /// Instantiate PrometheusMetrics struct
291    pub fn build(self) -> Result<PrometheusMetrics, Box<dyn std::error::Error>> {
292        let http_requests_total_opts =
293            Opts::new("http_requests_total", "Total number of HTTP requests")
294                .namespace(&self.namespace)
295                .const_labels(self.const_labels.clone());
296
297        let http_requests_total =
298            IntCounterVec::new(http_requests_total_opts, &["endpoint", "method", "status"])?;
299
300        let http_requests_duration_seconds_opts = HistogramOpts::new(
301            "http_requests_duration_seconds",
302            "HTTP request duration in seconds for all requests",
303        )
304        .namespace(&self.namespace)
305        .buckets(self.buckets.to_vec())
306        .const_labels(self.const_labels.clone());
307
308        let http_requests_duration_seconds = HistogramVec::new(
309            http_requests_duration_seconds_opts,
310            &["endpoint", "method", "status"],
311        )?;
312
313        self.registry
314            .register(Box::new(http_requests_total.clone()))?;
315        self.registry
316            .register(Box::new(http_requests_duration_seconds.clone()))?;
317
318        Ok(PrometheusMetrics {
319            http_requests_total,
320            http_requests_duration_seconds,
321            registry: self.registry,
322            namespace: self.namespace,
323            endpoint: self.endpoint,
324            const_labels: self.const_labels,
325        })
326    }
327}

329#[derive(Clone)]
330#[must_use = "must be set up as middleware for actix-web"]
331/// By default two metrics are tracked (this assumes the namespace `actix_web_prom`):
332///
333///   - `actix_web_prom_http_requests_total` (labels: endpoint, method, status): the total
334///   number of HTTP requests handled by the actix HttpServer.
335///
336///   - `actix_web_prom_http_requests_duration_seconds` (labels: endpoint, method,
337///    status): the request duration for all HTTP requests handled by the actix
338///    HttpServer.
339pub struct PrometheusMetrics {
340    pub(crate) http_requests_total: IntCounterVec,
341    pub(crate) http_requests_duration_seconds: HistogramVec,
342
343    /// exposed registry for custom prometheus metrics
344    pub registry: Registry,
345    pub(crate) namespace: String,
346    pub(crate) endpoint: Option<String>,
347    pub(crate) const_labels: HashMap<String, String>,
348}
349
350impl PrometheusMetrics {
351    fn metrics(&self) -> String {
352        let mut buffer = vec![];
353        TextEncoder::new()
354            .encode(&self.registry.gather(), &mut buffer)
355            .unwrap();
356        String::from_utf8(buffer).unwrap()
357    }
358
359    fn matches(&self, path: &str, method: &Method) -> bool {
360        if self.endpoint.is_some() {
361            self.endpoint.as_ref().unwrap() == path && method == Method::GET
362        } else {
363            false
364        }
365    }
366
367    fn update_metrics(&self, path: &str, method: &Method, status: StatusCode, clock: Instant) {
368        let method = method.to_string();
369        let status = status.as_u16().to_string();
370
371        let elapsed = clock.elapsed();
372        let duration =
373            (elapsed.as_secs() as f64) + f64::from(elapsed.subsec_nanos()) / 1_000_000_000_f64;
374        self.http_requests_duration_seconds
375            .with_label_values(&[path, &method, &status])
376            .observe(duration);
377
378        self.http_requests_total
379            .with_label_values(&[path, &method, &status])
380            .inc();
381    }
382}

405        where
406        S: Service<ServiceRequest>,
407    {
408        #[pin]
409        fut: S::Future,
410        time: Instant,
411        inner: Arc<PrometheusMetrics>,

415
416impl<S, B> Future for LoggerResponse<S>
417where
418    S: Service<ServiceRequest, Response = ServiceResponse<B>, Error = Error>,
419{
420    type Output = Result<ServiceResponse<EitherBody<StreamLog<B>, StreamLog<String>>>, Error>;
421
422    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
423        let this = self.project();
424
425        let res = match ready!(this.fut.poll(cx)) {
426            Ok(res) => res,
427            Err(e) => return Poll::Ready(Err(e)),
428        };
429
430        let time = *this.time;
431        let req = res.request();
432        let method = req.method().clone();
433        let pattern_or_path = req
434            .match_pattern()
435            .unwrap_or_else(|| req.path().to_string());
436        let path = req.path().to_string();
437        let inner = this.inner.clone();
438
439        Poll::Ready(Ok(res.map_body(move |mut head, body| {
440            // We short circuit the response status and body to serve the endpoint
441            // automagically. This way the user does not need to set the middleware *AND*
442            // an endpoint to serve middleware results. The user is only required to set
443            // the middleware and tell us what the endpoint should be.
444            if inner.matches(&path, &method) {
445                head.status = StatusCode::OK;
446                head.headers.insert(
447                    CONTENT_TYPE,
448                    HeaderValue::from_static("text/plain; version=0.0.4; charset=utf-8"),
449                );
450
451                EitherBody::right(StreamLog {
452                    body: inner.metrics(),
453                    size: 0,
454                    clock: time,
455                    inner,
456                    status: head.status,
457                    path: pattern_or_path,
458                    method,
459                })
460            } else {
461                EitherBody::left(StreamLog {
462                    body,
463                    size: 0,
464                    clock: time,
465                    inner,
466                    status: head.status,
467                    path: pattern_or_path,
468                    method,
469                })
470            }
471        })))
472    }
473}

525impl<B: MessageBody> MessageBody for StreamLog<B> {
526    type Error = B::Error;

530    }
531
532    fn poll_next(
533        self: Pin<&mut Self>,
534        cx: &mut Context<'_>,
535    ) -> Poll<Option<Result<Bytes, Self::Error>>> {
536        let this = self.project();
537        match ready!(this.body.poll_next(cx)) {
538            Some(Ok(chunk)) => {
539                *this.size += chunk.len();
540                Poll::Ready(Some(Ok(chunk)))
541            }
542            Some(Err(err)) => Poll::Ready(Some(Err(err))),
543            None => Poll::Ready(None),
544        }

848    async fn middleware_custom_registry_works() {
849        // Init Prometheus Registry
850        let registry = Registry::new();
851
852        let counter_opts = Opts::new("test_counter", "test counter help");
853        let counter = Counter::with_opts(counter_opts).unwrap();
854        registry.register(Box::new(counter.clone())).unwrap();
855
856        counter.inc_by(10_f64);
857
858        // Init PrometheusMetrics
859        let prometheus = PrometheusMetricsBuilder::new("actix_web_prom")
860            .registry(registry)
861            .endpoint("/metrics")
862            .build()
863            .unwrap();
864
865        let app = init_service(
866            App::new()
867                .wrap(prometheus.clone())
868                .service(web::resource("/test").to(|| async { HttpResponse::Ok().finish() })),
869        )
870        .await;
871
872        // all http counters are 0 because this is the first http request,
873        // so we should get only 10 on test counter
874        let response =
875            call_and_read_body(&app, TestRequest::with_uri("/metrics").to_request()).await;
876
877        let ten_test_counter =
878            "# HELP test_counter test counter help\n# TYPE test_counter counter\ntest_counter 10\n";
879        assert_eq!(
880            String::from_utf8(response.to_vec()).unwrap(),
881            ten_test_counter
882        );
883
884        // all http counters are 1 because this is the second http request,
885        // plus 10 on test counter
886        let response =
887            call_and_read_body(&app, TestRequest::with_uri("/metrics").to_request()).await;
888        let response_string = String::from_utf8(response.to_vec()).unwrap();
889
890        let one_http_counters = "# HELP actix_web_prom_http_requests_total Total number of HTTP requests\n# TYPE actix_web_prom_http_requests_total counter\nactix_web_prom_http_requests_total{endpoint=\"/metrics\",method=\"GET\",status=\"200\"} 1";
891
892        assert!(response_string.contains(ten_test_counter));
893        assert!(response_string.contains(one_http_counters));
894    }
